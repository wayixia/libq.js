// 处理客户端请求数据
if(isset($_POST['postdata']) {
  $postdata = urldecode($_POST['postdata'])

Examples

元素自身可拖拽初始化

Q.ready(function() {
   Q.drag({
     id: 'test-drag',  // 元素id
     self: true   // 元素自身任何区域可以拖拽
   });
});

自定义内部子节点触发的拖拽初始化 

Q.ready(function() {
   Q.drag({
     id: 'test-drag',  // 元素id
     self: true        // 元素自身任何区域可以拖拽
     objects: ['test-drag-child1', 'test-drag-child2']  // test-drag-child1, test-drag-child2 都为test-drag元素的子节点元素
   });
});
